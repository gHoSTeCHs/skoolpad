```php
public function isAutoGradable(QuestionType $type): bool
{
    return in_array($type, [
        QuestionType::Mcq,
        QuestionType::MultiSelectMcq,
        QuestionType::TrueFalse,
        QuestionType::NumericEntry,
        QuestionType::AssertionReason,
        QuestionType::Cloze,
        QuestionType::Matching,
        QuestionType::Ordering,
        QuestionType::FillBlank,
        QuestionType::DiagramLabel,
        QuestionType::MatrixMatching,
    ]);
}

public function gradeAnswer(Question $question, array $responseData): ?bool
{
    if (! $this->isAutoGradable($question->question_type)) {
        return null;
    }

    return match ($question->question_type) {
        QuestionType::Mcq => $this->gradeMcq($question, $responseData),
        QuestionType::MultiSelectMcq => $this->gradeMultiSelect($question, $responseData),
        QuestionType::TrueFalse => $this->gradeTrueFalse($question, $responseData),
        QuestionType::NumericEntry => $this->gradeNumericEntry($question, $responseData),
        QuestionType::AssertionReason => $this->gradeAssertionReason($question, $responseData),
        QuestionType::Cloze => $this->gradeCloze($question, $responseData),
        QuestionType::Matching => $this->gradeMatching($question, $responseData),
        QuestionType::Ordering => $this->gradeOrdering($question, $responseData),
        QuestionType::FillBlank => $this->gradeFillBlank($question, $responseData),
        QuestionType::DiagramLabel => $this->gradeDiagramLabel($question, $responseData),
        QuestionType::MatrixMatching => $this->gradeMatrixMatching($question, $responseData),
        default => null,
    };
}
```

Add these private methods (after the existing `gradeMcq`):

```php
private function gradeMultiSelect(Question $question, array $responseData): bool
{
    $selected = collect($responseData['selected_labels'] ?? [])->sort()->values()->toArray();
    $correct = collect($question->response_config['options'] ?? [])
        ->where('is_correct', true)
        ->pluck('label')
        ->sort()
        ->values()
        ->toArray();

    return $selected === $correct;
}

private function gradeTrueFalse(Question $question, array $responseData): bool
{
    return ($responseData['answer'] ?? null) === ($question->response_config['correct_answer'] ?? null);
}

private function gradeNumericEntry(Question $question, array $responseData): bool
{
    $value = $responseData['value'] ?? null;
    $correct = $question->response_config['answer'] ?? null;

    if ($value === null || $correct === null) {
        return false;
    }

    $tolerance = (float) ($question->response_config['tolerance'] ?? 0);

    return abs((float) $value - (float) $correct) <= $tolerance;
}

private function gradeAssertionReason(Question $question, array $responseData): bool
{
    $selected = $responseData['selected'] ?? null;
    $correctOption = collect($question->response_config['options'] ?? [])->firstWhere('is_correct', true);

    return $correctOption && $selected === $correctOption['label'];
}

private function gradeCloze(Question $question, array $responseData): bool
{
    $studentGaps = $responseData['gaps'] ?? [];
    $configGaps = $question->response_config['gaps'] ?? [];

    if (empty($configGaps)) {
        return false;
    }

    foreach ($configGaps as $gap) {
        $position = (string) $gap['position'];

        if ((int) ($studentGaps[$position] ?? -1) !== (int) $gap['correct']) {
            return false;
        }
    }

    return true;
}

private function gradeMatching(Question $question, array $responseData): bool
{
    $studentPairs = $responseData['pairs'] ?? [];
    $configPairs = $question->response_config['pairs'] ?? [];

    if (empty($configPairs)) {
        return false;
    }

    foreach (array_keys($configPairs) as $index) {
        if ((int) ($studentPairs[(string) $index] ?? -1) !== (int) $index) {
            return false;
        }
    }

    return true;
}

private function gradeOrdering(Question $question, array $responseData): bool
{
    $student = array_map('intval', $responseData['order'] ?? []);
    $correct = array_map('intval', $question->response_config['correct_order'] ?? []);

    return $student === $correct;
}

private function gradeFillBlank(Question $question, array $responseData): bool
{
    $studentBlanks = $responseData['blanks'] ?? [];
    $configBlanks = $question->response_config['blanks'] ?? [];
    $caseSensitive = $question->response_config['case_sensitive'] ?? false;

    if (empty($configBlanks)) {
        return false;
    }

    $correct = 0;

    foreach ($configBlanks as $blank) {
        $position = (string) $blank['position'];
        $studentAnswer = $studentBlanks[$position] ?? '';

        foreach ($blank['correct_answers'] ?? [] as $accepted) {
            $matches = $caseSensitive
                ? $studentAnswer === $accepted
                : strtolower(trim($studentAnswer)) === strtolower(trim($accepted));

            if ($matches) {
                $correct++;
                break;
            }
        }
    }

    return ($correct / count($configBlanks)) >= 0.8;
}

private function gradeDiagramLabel(Question $question, array $responseData): bool
{
    $studentLabels = $responseData['labels'] ?? [];
    $configLabels = $question->response_config['labels'] ?? [];

    if (empty($configLabels)) {
        return false;
    }

    foreach ($configLabels as $index => $label) {
        $key = 'hotspot_' . $index;
        $student = strtolower(trim($studentLabels[$key] ?? ''));
        $correct = strtolower(trim($label['answer'] ?? ''));

        if ($student !== $correct) {
            return false;
        }
    }

    return true;
}

private function gradeMatrixMatching(Question $question, array $responseData): bool
{
    $studentMatches = $responseData['matches'] ?? [];
    $correctMapping = $question->response_config['mapping'] ?? [];

    if (empty($correctMapping)) {
        return false;
    }

    foreach ($correctMapping as $leftIndex => $rightIndices) {
        $studentRight = array_map('intval', $studentMatches[(string) $leftIndex] ?? []);
        $correctRight = array_map('intval', (array) $rightIndices);
        sort($studentRight);
        sort($correctRight);

        if ($studentRight !== $correctRight) {
            return false;
        }
    }

    return true;
}
```

```tsx
import { useState } from 'react';

import { cn } from '@/lib/utils';
import type { MultiSelectMcqConfig } from '@/types/questions';

interface MultiSelectInputProps {
    responseConfig: MultiSelectMcqConfig;
    onSubmit: (data: { selected_labels: string[] }) => void;
    feedback?: { isCorrect: boolean | null; correctAnswer: { correct_labels?: string[] } | null } | null;
    readOnly?: boolean;
    existingAnswer?: { selected_labels: string[] } | null;
}

export function MultiSelectInput({ responseConfig, onSubmit, feedback, readOnly, existingAnswer }: MultiSelectInputProps) {
    const [selectedLabels, setSelectedLabels] = useState<string[]>(existingAnswer?.selected_labels ?? []);
    const options = responseConfig?.options ?? [];
    const isSubmitted = !!feedback || !!readOnly;
    const correctLabels = (feedback?.correctAnswer as { correct_labels?: string[] })?.correct_labels ?? [];
    const displaySelected = existingAnswer?.selected_labels ?? selectedLabels;

    function toggleLabel(label: string) {
        if (isSubmitted) return;
        setSelectedLabels((prev) => (prev.includes(label) ? prev.filter((l) => l !== label) : [...prev, label]));
    }

    function getOptionStyle(label: string, isCorrectOption: boolean) {
        if (!isSubmitted) {
            return selectedLabels.includes(label)
                ? 'border-primary bg-primary/5 ring-2 ring-primary/20'
                : 'border-border hover:border-primary/40 hover:bg-accent/50';
        }

        const studentSelected = displaySelected.includes(label);
        const isInCorrect = correctLabels.includes(label) || isCorrectOption;

        if (isInCorrect && studentSelected) return 'border-emerald-500 bg-emerald-500/10 text-emerald-700 dark:text-emerald-400 reader:text-emerald-400';
        if (isInCorrect && !studentSelected) return 'border-emerald-500 bg-emerald-500/5 ring-1 ring-emerald-500/40';
        if (studentSelected && !isInCorrect) return 'border-destructive bg-destructive/10 text-destructive';
        return 'border-border opacity-50';
    }

    return (
        <div className="space-y-3">
            <p className="text-xs text-muted-foreground" style={{ fontFamily: 'var(--font-body)' }}>
                Select all that apply
            </p>
            <div className="space-y-2">
                {options.map((option) => (
                    <button
                        key={option.label}
                        type="button"
                        onClick={() => toggleLabel(option.label)}
                        disabled={isSubmitted}
                        className={cn(
                            'flex w-full items-start gap-3 rounded-lg border p-3 text-left transition-all',
                            getOptionStyle(option.label, option.is_correct),
                            !isSubmitted && 'cursor-pointer',
                        )}
                    >
                        <span
                            className={cn(
                                'mt-0.5 flex h-5 w-5 shrink-0 items-center justify-center rounded border-2 text-xs font-bold',
                                displaySelected.includes(option.label)
                                    ? 'border-primary bg-primary text-primary-foreground'
                                    : 'border-muted-foreground/50',
                            )}
                        >
                            {displaySelected.includes(option.label) && '✓'}
                        </span>
                        <span className="text-sm leading-relaxed" style={{ fontFamily: 'var(--font-content)' }}>
                            <span className="mr-2 font-bold">{option.label}.</span>
                            {option.text}
                        </span>
                    </button>
                ))}
            </div>
            {!isSubmitted && selectedLabels.length > 0 && (
                <button
                    type="button"
                    onClick={() => onSubmit({ selected_labels: selectedLabels })}
                    className="rounded-md bg-primary px-4 py-2 text-sm font-medium text-primary-foreground shadow-xs transition-colors hover:bg-primary/90"
                >
                    Submit Answer
                </button>
            )}
        </div>
    );
```

```tsx
interface TrueFalseInputProps {
    responseConfig: TrueFalseConfig;
    onSubmit: (data: { answer: boolean; justification?: string }) => void;
    feedback?: { isCorrect: boolean | null; correctAnswer: { correct_answer?: boolean } | null } | null;
    readOnly?: boolean;
    existingAnswer?: { answer: boolean; justification?: string } | null;
}

export function TrueFalseInput({ responseConfig, onSubmit, feedback, readOnly, existingAnswer }: TrueFalseInputProps) {
    const [answer, setAnswer] = useState<boolean | null>(existingAnswer?.answer ?? null);
    const [justification, setJustification] = useState(existingAnswer?.justification ?? '');
    const isSubmitted = !!feedback || !!readOnly;
    const requiresJustification = responseConfig?.requires_justification ?? false;
    const correctAnswer = (feedback?.correctAnswer as { correct_answer?: boolean })?.correct_answer;
    const displayAnswer = existingAnswer?.answer ?? answer;

    function getButtonStyle(value: boolean) {
        if (!isSubmitted) {
            return answer === value
                ? 'border-primary bg-primary/5 ring-2 ring-primary/20'
                : 'border-border hover:border-primary/40 hover:bg-accent/50';
        }

        const studentChose = displayAnswer === value;
        const isCorrectChoice = correctAnswer === value;

        if (isCorrectChoice) return 'border-emerald-500 bg-emerald-500/10 text-emerald-700 dark:text-emerald-400';
        if (studentChose) return 'border-destructive bg-destructive/10 text-destructive';
        return 'border-border opacity-50';
    }

    return (
        <div className="space-y-3">
            <div className="grid grid-cols-2 gap-3">
                {([true, false] as const).map((value) => (
                    <button
                        key={String(value)}
                        type="button"
                        onClick={() => !isSubmitted && setAnswer(value)}
                        disabled={isSubmitted}
                        className={cn(
                            'rounded-lg border p-4 text-center text-sm font-semibold transition-all',
                            getButtonStyle(value),
                            !isSubmitted && 'cursor-pointer',
                        )}

                        {value ? 'True' : 'False'}
                    </button>
                ))}

                    value={justification}

                    style={{ fontFamily: 'var(--font-content)' }}
```

```tsx
interface OrderingInputProps {
    responseConfig: OrderingConfig;
    onSubmit: (data: { order: number[] }) => void;
    feedback?: { isCorrect: boolean | null; correctAnswer: { correct_order?: number[]; items?: string[] } | null } | null;
    readOnly?: boolean;
    existingAnswer?: { order: number[] } | null;
}

export function OrderingInput({ responseConfig, onSubmit, feedback, readOnly, existingAnswer }: OrderingInputProps) {
    const items = responseConfig?.items ?? [];
    const correctOrder = responseConfig?.correct_order ?? [];
    const [order, setOrder] = useState<number[]>(existingAnswer?.order ?? items.map((_, i) => i));
    const isSubmitted = !!feedback || !!readOnly;

    const sensors = useSensors(
        useSensor(PointerSensor),
        useSensor(KeyboardSensor, { coordinateGetter: sortableKeyboardCoordinates }),
    );

    function handleDragEnd(event: DragEndEvent) {
        const { active, over } = event;
        if (!over || active.id === over.id) return;
        const activeIdx = order.findIndex((_, i) => `item-${i}` === active.id);
        const overIdx = order.findIndex((_, i) => `item-${i}` === over.id);
        setOrder((prev) => arrayMove(prev, activeIdx, overIdx));
    }

    function isCorrectPosition(displayIndex: number): boolean | null {
        if (!isSubmitted) return null;
        return order[displayIndex] === correctOrder[displayIndex];
    }

    return (
        <div className="space-y-3">
            {!isSubmitted && (
                <p className="text-xs text-muted-foreground" style={{ fontFamily: 'var(--font-body)' }}>
                    Drag to arrange in the correct order
                </p>
            )}
            <DndContext sensors={sensors} collisionDetection={closestCenter} onDragEnd={handleDragEnd}>
                <SortableContext items={order.map((_, i) => `item-${i}`)} strategy={verticalListSortingStrategy}>
                    <div className="space-y-2">
                        {order.map((itemIndex, displayIndex) => (
                            <SortableItem
                                key={`item-${displayIndex}`}
                                id={`item-${displayIndex}`}
                                text={items[itemIndex] ?? ''}
                                displayIndex={displayIndex}
                                isCorrect={isCorrectPosition(displayIndex)}
                                isSubmitted={isSubmitted}
                            />
                        ))}
                    </div>
                </SortableContext>
            </DndContext>
            {!isSubmitted && (
                <button
                    type="button"
                    onClick={() => onSubmit({ order })}
                    className="rounded-md bg-primary px-4 py-2 text-sm font-medium text-primary-foreground shadow-xs transition-colors hover:bg-primary/90"
                >
                    Submit Answer
                </button>
            )}
        </div>
    );
}
```

```tsx
                    </svg>
                </button>
            )}
            <span style={{ fontFamily: 'var(--font-content)' }}>{text}</span>
        </div>
    );
}

interface MatchingInputProps {
    responseConfig: MatchingConfig;
    onSubmit: (data: { pairs: Record<string, number> }) => void;
    feedback?: { isCorrect: boolean | null; correctAnswer: { pairs?: { left: string; right: string }[] } | null } | null;
    readOnly?: boolean;
    existingAnswer?: { pairs: Record<string, number> } | null;
}

export function MatchingInput({ responseConfig, onSubmit, feedback, readOnly, existingAnswer }: MatchingInputProps) {
    const pairs = responseConfig?.pairs ?? [];
    const leftItems = pairs.map((p) => p.left);

    function buildInitialOrder(): number[] {
        if (existingAnswer?.pairs) {
            return leftItems.map((_, i) => existingAnswer.pairs[String(i)] ?? i);
        }
        const indices = pairs.map((_, i) => i);
        for (let i = indices.length - 1; i > 0; i--) {
            const j = Math.floor(Math.random() * (i + 1));
            [indices[i], indices[j]] = [indices[j], indices[i]];
        }
        return indices;
    }

    const [rightOrder, setRightOrder] = useState<number[]>(buildInitialOrder);
    const isSubmitted = !!feedback || !!readOnly;

    const sensors = useSensors(
        useSensor(PointerSensor),
        useSensor(KeyboardSensor, { coordinateGetter: sortableKeyboardCoordinates }),
    );

    function handleDragEnd(event: DragEndEvent) {
        const { active, over } = event;
        if (!over || active.id === over.id) return;
        const activeIdx = rightOrder.findIndex((_, i) => `right-${i}` === active.id);
        const overIdx = rightOrder.findIndex((_, i) => `right-${i}` === over.id);
        setRightOrder((prev) => arrayMove(prev, activeIdx, overIdx));
    }

    function handleSubmit() {
        const pairsData: Record<string, number> = {};
        rightOrder.forEach((originalIdx, position) => {
            pairsData[String(position)] = originalIdx;
        });
        onSubmit({ pairs: pairsData });
    }

    return (
        <div className="space-y-3">
            {!isSubmitted && (
                <p className="text-xs text-muted-foreground" style={{ fontFamily: 'var(--font-body)' }}>
                    Drag the right column to match each item on the left
                </p>
            )}
            <div className="grid grid-cols-2 gap-3">
                <div className="space-y-2">
                    {leftItems.map((item, i) => (
                        <div
                            key={i}
                            className="flex items-center rounded-lg border border-border bg-muted/30 p-3 text-sm"
                            style={{ fontFamily: 'var(--font-content)' }}
                        >
                            <span className="mr-2 text-xs font-bold text-muted-foreground">{i + 1}.</span>
                            {item}
                        </div>
                    ))}
                </div>
                <DndContext sensors={sensors} collisionDetection={closestCenter} onDragEnd={handleDragEnd}>
                    <SortableContext items={rightOrder.map((_, i) => `right-${i}`)} strategy={verticalListSortingStrategy}>
                        <div className="space-y-2">
                            {rightOrder.map((originalIdx, displayIdx) => (
                                <SortableRightItem
                                    key={`right-${displayIdx}`}
                                    id={`right-${displayIdx}`}
                                    text={pairs[originalIdx]?.right ?? ''}
                                    isCorrect={isSubmitted ? originalIdx === displayIdx : null}
                                    isSubmitted={isSubmitted}
                                />
                            ))}
                        </div>
                    </SortableContext>
                </DndContext>
            </div>
            {!isSubmitted && (
                <button
                    type="button"
                    onClick={handleSubmit}
                    className="rounded-md bg-primary px-4 py-2 text-sm font-medium text-primary-foreground shadow-xs transition-colors hover:bg-primary/90"
                >
                    Submit Answer

    );
}
```

```tsx
import { useState } from 'react';

import { cn } from '@/lib/utils';
import type { FillBlankConfig } from '@/types/questions';

interface FillBlankInputProps {
    responseConfig: FillBlankConfig;
    onSubmit: (data: { blanks: Record<string, string> }) => void;
    feedback?: { isCorrect: boolean | null; correctAnswer: { blanks?: { position: number; correct_answers: string[] }[] } | null } | null;
    readOnly?: boolean;
    existingAnswer?: { blanks: Record<string, string> } | null;
}

export function FillBlankInput({ responseConfig, onSubmit, feedback, readOnly, existingAnswer }: FillBlankInputProps) {
    const blanks = responseConfig?.blanks ?? [];
    const [values, setValues] = useState<Record<string, string>>(existingAnswer?.blanks ?? {});
    const isSubmitted = !!feedback || !!readOnly;
    const displayValues = existingAnswer?.blanks ?? values;
    const correctBlanks = (feedback?.correctAnswer as { blanks?: { position: number; correct_answers: string[] }[] })?.blanks ?? [];

    function getInputStyle(position: number) {
        if (!isSubmitted) return 'border-border focus:ring-primary/20 focus:border-primary';
        const correct = correctBlanks.find((b) => b.position === position);
        if (!correct) return 'border-border';
        const studentVal = (displayValues[String(position)] ?? '').toLowerCase().trim();
        const isCorrect = correct.correct_answers.some((a) => a.toLowerCase().trim() === studentVal);
        return isCorrect ? 'border-emerald-500 bg-emerald-500/10' : 'border-destructive bg-destructive/10';
    }

    const allFilled = blanks.every((b) => (values[String(b.position)] ?? '').trim() !== '');

    return (
        <div className="space-y-4">
            <div className="space-y-3">
                {blanks.map((blank, i) => {
                    const correctAnswers = correctBlanks.find((b) => b.position === blank.position)?.correct_answers ?? [];
                    return (
                        <div key={blank.position} className="flex items-center gap-3">
                            <span className="w-8 shrink-0 text-right text-xs font-semibold text-muted-foreground" style={{ fontFamily: 'var(--font-body)' }}>
                                {i + 1}.
                            </span>
                            <input
                                type="text"
                                value={displayValues[String(blank.position)] ?? ''}
                                onChange={(e) => setValues((prev) => ({ ...prev, [String(blank.position)]: e.target.value }))}
                                disabled={isSubmitted}
                                placeholder={`Blank ${i + 1}`}
                                className={cn(
                                    'flex-1 rounded-md border px-3 py-2 text-sm focus:outline-none focus:ring-2 disabled:opacity-60',
                                    getInputStyle(blank.position),
                                )}
                                style={{ fontFamily: 'var(--font-content)' }}
                            />
                            {isSubmitted && correctAnswers.length > 0 && (
                                <span className="text-xs text-emerald-600 dark:text-emerald-400" style={{ fontFamily: 'var(--font-body)' }}>
                                    {correctAnswers[0]}
                                </span>
                            )}
                        </div>
                    );
                })}
            </div>
            {!isSubmitted && allFilled && (
                <button
                    type="button"
                    onClick={() => onSubmit({ blanks: values })}
                    className="rounded-md bg-primary px-4 py-2 text-sm font-medium text-primary-foreground shadow-xs transition-colors hover:bg-primary/90"
                >
                    Submit Answer
                </button>
            )}
        </div>
    );
}
```

```tsx
import { useState } from 'react';

import { cn } from '@/lib/utils';
import type { ClozeConfig } from '@/types/questions';

interface ClozeInputProps {
    responseConfig: ClozeConfig;
    onSubmit: (data: { gaps: Record<string, number> }) => void;
    feedback?: { isCorrect: boolean | null; correctAnswer: { gaps?: { position: number; options: string[]; correct: number }[] } | null } | null;
    readOnly?: boolean;
    existingAnswer?: { gaps: Record<string, number> } | null;
}

export function ClozeInput({ responseConfig, onSubmit, feedback, readOnly, existingAnswer }: ClozeInputProps) {

    const [values, setValues] = useState<Record<string, number>>(existingAnswer?.gaps ?? {});
    const isSubmitted = !!feedback || !!readOnly;
    const displayValues = existingAnswer?.gaps ?? values;

    function getSelectStyle(position: number) {
        if (!isSubmitted) return 'border-border';
        const correct = correctGaps.find((g) => g.position === position);
        if (!correct) return 'border-border';
        return (displayValues[String(position)] ?? -1) === correct.correct
            ? 'border-emerald-500 bg-emerald-500/10'
```

```tsx
import { useState } from 'react';

import { cn } from '@/lib/utils';
import type { MatrixMatchingConfig } from '@/types/questions';

interface MatrixMatchingInputProps {
    responseConfig: MatrixMatchingConfig;
    onSubmit: (data: { matches: Record<string, number[]> }) => void;
    feedback?: { isCorrect: boolean | null; correctAnswer: { mapping?: Record<number, number[]>; left?: string[]; right?: string[] } | null } | null;
    readOnly?: boolean;
    existingAnswer?: { matches: Record<string, number[]> } | null;
}

export function MatrixMatchingInput({ responseConfig, onSubmit, feedback, readOnly, existingAnswer }: MatrixMatchingInputProps) {
    const left = responseConfig?.left ?? [];
    const right = responseConfig?.right ?? [];
    const correctMapping = (feedback?.correctAnswer as { mapping?: Record<number, number[]> })?.mapping ?? {};

    const [matches, setMatches] = useState<Record<string, number[]>>(existingAnswer?.matches ?? {});
    const isSubmitted = !!feedback || !!readOnly;
    const displayMatches = existingAnswer?.matches ?? matches;

    function toggleCell(leftIdx: number, rightIdx: number) {
        if (isSubmitted) return;
        setMatches((prev) => {
            const current = prev[String(leftIdx)] ?? [];
            return {
                ...prev,
                [String(leftIdx)]: current.includes(rightIdx)
                    ? current.filter((i) => i !== rightIdx)
                    : [...current, rightIdx],
            };
        });
    }

    function isCellCorrect(leftIdx: number, rightIdx: number): boolean | null {
        if (!isSubmitted) return null;
        const correctRight = (correctMapping[leftIdx] ?? []).map(Number);
        return correctRight.includes(rightIdx);
    }

    function getCellStyle(leftIdx: number, rightIdx: number) {
        const isChecked = (displayMatches[String(leftIdx)] ?? []).includes(rightIdx);
        const correct = isCellCorrect(leftIdx, rightIdx);
        if (!isSubmitted) {
            return isChecked ? 'bg-primary/10 border-primary' : 'border-border hover:bg-accent/50';
        }
        if (correct === true && isChecked) return 'bg-emerald-500/10 border-emerald-500';
        if (correct === true && !isChecked) return 'bg-emerald-500/5 border-emerald-500/50 ring-1 ring-emerald-500/40';
        if (correct === false && isChecked) return 'bg-destructive/10 border-destructive';
        return 'border-border';
    }

    return (
        <div className="space-y-4">
            <div className="overflow-x-auto">
                <table className="w-full border-collapse">
                    <thead>
                        <tr>
                            <th className="border border-border bg-muted/40 p-2 text-left text-xs font-semibold text-muted-foreground" style={{ fontFamily: 'var(--font-body)' }}>
                            </th>
                            {right.map((item, j) => (
                                <th key={j} className="border border-border bg-muted/40 p-2 text-center text-xs font-semibold" style={{ fontFamily: 'var(--font-body)' }}>
                                    {item}
                                </th>
                            ))}
                        </tr>
                    </thead>
                    <tbody>
                        {left.map((item, i) => (
                            <tr key={i}>
                                <td className="border border-border bg-muted/20 p-2 text-xs font-medium" style={{ fontFamily: 'var(--font-content)' }}>
                                    {item}
                                </td>
                                {right.map((_, j) => {
                                    const isChecked = (displayMatches[String(i)] ?? []).includes(j);
                                    return (
                                        <td key={j} className="border border-border p-1 text-center">
                                            <button
                                                type="button"
                                                onClick={() => toggleCell(i, j)}
                                                disabled={isSubmitted}
                                                className={cn(
                                                    'mx-auto flex h-7 w-7 items-center justify-center rounded border text-xs font-bold transition-all',
                                                    getCellStyle(i, j),
                                                    !isSubmitted && 'cursor-pointer',
                                                )}
                                                aria-label={`Match ${item} with ${right[j]}`}
                                            >
                                                {isChecked && '✓'}
                                            </button>
                                        </td>
                                    );
                                })}
                            </tr>
                        ))}
                    </tbody>
                </table>
            </div>
            {!isSubmitted && (
                <button
                    type="button"
                    onClick={() => onSubmit({ matches })}
                    className="rounded-md bg-primary px-4 py-2 text-sm font-medium text-primary-foreground shadow-xs transition-colors hover:bg-primary/90"
                >
                    Submit Answer
                </button>
            )}
        </div>
    );
}
```

### Step 1: Create `group-renderer.tsx`

For Phase C, the group renderer shows child questions sequentially with their own `QuestionAnswerInput` components. Each child's answer is tracked locally. When the student submits all children (or skips), the group calls `onSubmit` with all child answers bundled into `response_data`. This is stored as a single non-gradable practice answer.

```tsx
import { useState } from 'react';

import { QuestionAnswerInput } from './question-answer-input';
import type { PracticeQuestionData } from '@/types/practice';

interface GroupRendererProps {
    children: PracticeQuestionData['children'];
    onSubmit: (data: { group_answers: Record<string, Record<string, unknown>> }) => void;
    feedback?: { isCorrect: boolean | null; correctAnswer: Record<string, unknown> | null } | null;
    readOnly?: boolean;
    existingAnswer?: { group_answers: Record<string, Record<string, unknown>> } | null;
}

export function GroupRenderer({ children, onSubmit, feedback, readOnly, existingAnswer }: GroupRendererProps) {
    const [childAnswers, setChildAnswers] = useState<Record<string, Record<string, unknown>>>(existingAnswer?.group_answers ?? {});
    const isSubmitted = !!feedback || !!readOnly;
    const allAnswered = children.every((child) => childAnswers[child.id] !== undefined);

    function handleChildAnswer(childId: string, data: Record<string, unknown>) {
        setChildAnswers((prev) => ({ ...prev, [childId]: data }));
    }

    return (
        <div className="space-y-6">
            {children.map((child, index) => (
                <div key={child.id} className="rounded-lg border border-border/60 p-4">
                    <div className="mb-3 flex items-center gap-2">
                        <span className="flex h-6 w-6 shrink-0 items-center justify-center rounded-full bg-muted text-xs font-bold text-muted-foreground">
                            {index + 1}
                        </span>
                        <p className="text-sm leading-relaxed" style={{ fontFamily: 'var(--font-content)' }}>
                            {child.content}
                        </p>
                    </div>
                    <QuestionAnswerInput
                        questionType={child.question_type}
                        responseConfig={child.response_config}
                        onSubmit={(data) => handleChildAnswer(child.id, data)}
                        readOnly={isSubmitted || childAnswers[child.id] !== undefined}
                        existingAnswer={(existingAnswer?.group_answers?.[child.id] ?? childAnswers[child.id]) as Record<string, unknown> | null}
                    />
                </div>
            ))}
            {!isSubmitted && allAnswered && (
                <button
                    type="button"
                    onClick={() => onSubmit({ group_answers: childAnswers })}
                    className="rounded-md bg-primary px-4 py-2 text-sm font-medium text-primary-foreground shadow-xs transition-colors hover:bg-primary/90"
                >
                    Submit All
                </button>
            )}
        </div>
    );
}
```

```tsx
import type {
    AssertionReasonConfig,
    CalculationConfig,
    ClozeConfig,
    DiagramLabelConfig,
    FillBlankConfig,
    MatchingConfig,
    MatrixMatchingConfig,
    McqConfig,
    MultiSelectMcqConfig,
    NumericEntryConfig,
    OrderingConfig,
    QuestionType,
    ResponseConfig,
    TrueFalseConfig,
} from '@/types/questions';

import type { PracticeQuestionData } from '@/types/practice';

import { AssertionReasonInput } from './assertion-reason-input';
import { CalculationInput } from './calculation-input';
import { ClozeInput } from './cloze-input';
import { DiagramLabelInput } from './diagram-label-input';
import { EssayInput } from './essay-input';
import { FillBlankInput } from './fill-blank-input';
import { GroupRenderer } from './group-renderer';
import { MatchingInput } from './matching-input';
import { MatrixMatchingInput } from './matrix-matching-input';
import { McqInput } from './mcq-input';
import { MultiSelectInput } from './multi-select-input';
import { NumericEntryInput } from './numeric-entry-input';
import { OrderingInput } from './ordering-input';
import { ShortAnswerInput } from './short-answer-input';
import { TheoryInput } from './theory-input';
import { TrueFalseInput } from './true-false-input';

interface QuestionAnswerInputProps {
    questionType: QuestionType;
    responseConfig: ResponseConfig;
    onSubmit: (data: Record<string, unknown>) => void;
    feedback?: { isCorrect: boolean | null; correctAnswer: Record<string, unknown> | null } | null;
    readOnly?: boolean;
    existingAnswer?: Record<string, unknown> | null;
    mediaUrl?: string;
    children?: PracticeQuestionData['children'];
}

export function QuestionAnswerInput({
    questionType,
    responseConfig,
    onSubmit,
    feedback,
    readOnly,
    existingAnswer,
    mediaUrl,
    children,
}: QuestionAnswerInputProps) {
    const common = { onSubmit, feedback: feedback as never, readOnly, existingAnswer: existingAnswer as never };

    switch (questionType) {
        case 'mcq':
            return <McqInput responseConfig={responseConfig as McqConfig} {...common} />;
        case 'multi_select_mcq':
            return <MultiSelectInput responseConfig={responseConfig as MultiSelectMcqConfig} {...common} />;
        case 'true_false':
            return <TrueFalseInput responseConfig={responseConfig as TrueFalseConfig} {...common} />;
        case 'numeric_entry':
            return <NumericEntryInput responseConfig={responseConfig as NumericEntryConfig} {...common} />;
        case 'assertion_reason':
            return <AssertionReasonInput responseConfig={responseConfig as AssertionReasonConfig} {...common} />;
        case 'ordering':
            return <OrderingInput responseConfig={responseConfig as OrderingConfig} {...common} />;
        case 'matching':
            return <MatchingInput responseConfig={responseConfig as MatchingConfig} {...common} />;
        case 'fill_blank':
            return <FillBlankInput responseConfig={responseConfig as FillBlankConfig} {...common} />;
        case 'cloze':
            return <ClozeInput responseConfig={responseConfig as ClozeConfig} {...common} />;
        case 'short_answer':
            return <ShortAnswerInput onSubmit={onSubmit} feedback={feedback} readOnly={readOnly} existingAnswer={existingAnswer as { text: string } | null} />;
        case 'theory':
            return <TheoryInput onSubmit={onSubmit} feedback={feedback} readOnly={readOnly} existingAnswer={existingAnswer as { text: string } | null} />;
        case 'essay':
            return <EssayInput onSubmit={onSubmit} feedback={feedback} readOnly={readOnly} existingAnswer={existingAnswer as { text: string } | null} />;
        case 'calculation':
            return <CalculationInput responseConfig={responseConfig as CalculationConfig} onSubmit={onSubmit} feedback={feedback} readOnly={readOnly} existingAnswer={existingAnswer as { value: string; unit: string; working: string } | null} />;
        case 'diagram_label':
            return <DiagramLabelInput responseConfig={responseConfig as DiagramLabelConfig} mediaUrl={mediaUrl} {...common} />;
        case 'matrix_matching':
            return <MatrixMatchingInput responseConfig={responseConfig as MatrixMatchingConfig} {...common} />;
        case 'group':
            return (
                <GroupRenderer
                    children={children ?? []}
                    onSubmit={onSubmit}
                    feedback={feedback}
                    readOnly={readOnly}
                    existingAnswer={existingAnswer as { group_answers: Record<string, Record<string, unknown>> } | null}
                />
            );
        default:
            return (
                <div className="rounded-lg border border-dashed border-border p-6 text-center">
                    <p className="text-sm text-muted-foreground" style={{ fontFamily: 'var(--font-body)' }}>
                        This question type ({questionType}) is not yet supported for practice.
                    </p>
                </div>
            );
    }
}
```
